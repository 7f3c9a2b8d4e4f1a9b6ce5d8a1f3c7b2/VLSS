```move
    coins: Coin<T>,

    // Check that the amount of coins to be burned is not zero

    let publish_ticket = PublishTicket {
        burn_token: token_messenger_minter::token_utils::calculate_token_id<T>(),
        mint_recipient,
        amount: coins.value(),
        destination_domain,
        destination_caller,
        wormhole_nonce,
        payload,
    };
    [...]
}
```
